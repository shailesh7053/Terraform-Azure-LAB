#Step 1 – Create Storage Account
resource "azurerm_storage_account" "fslogix_sa" {
  name                     = "fslogixdemostg01"   # must be globally unique
  resource_group_name      = azurerm_resource_group.demo_rg.name
  location                 = azurerm_resource_group.demo_rg.location
  account_tier             = "Standard"
  account_replication_type = "LRS"

  min_tls_version          = "TLS1_2"
}

##Step 2 – Create File Share
resource "azurerm_storage_share" "fslogix_share" {
  name                 = "fslogix"
  storage_account_name = azurerm_storage_account.fslogix_sa.name
  quota                = 100
}
